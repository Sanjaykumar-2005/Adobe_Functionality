python-docx==1.1.2     # docx inspection / generation + native pdf->word rebuild
pdfplumber==0.11.10    # table extraction for the native pdf->word engine
pdf2docx==0.5.8        # pdf -> word (legacy/alternate engine, still in conversion_service)
reportlab==4.2.2       # overlays: watermark text, page numbers, signatures

# ---- OCR (requires Tesseract + Ghostscript on the host) ----
ocrmypdf==16.4.3       # searchable pdf, preserves layout  (optional at runtime)
pytesseract==0.3.13    # lightweight fallback OCR          (optional at runtime)

# NOTE:
#   - LibreOffice (`soffice`) is the PRIMARY engine for BOTH Word->PDF and PDF->Word.
#     It is a system package, not a pip one — the Dockerfile installs it. Set
#     LIBREOFFICE_BIN to point at a non-standard install.
#     Without it: Word->PDF falls back to MS Word (Windows) then reportlab (text-only);
#     PDF->Word falls back to MS Word then the native services/pdf2word engine.
#     Nothing here is a hard requirement to boot the app.
#   - OCR requires the Tesseract binary; the app degrades gracefully with a clear error if absent.
